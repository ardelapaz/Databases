    a)  SELECT first_name, last_name
        FROM guests AS g
        FULL JOIN booking AS b ON b.guest_id = g.id
        WHERE b.guest_id IS NULL;

    b)  SELECT first_name, last_name, check_in_date, COUNT(*) count
        FROM guest AS g 
        FULL JOIN booking AS b ON b.guest_id = g.id
        GROUP BY check_in_date, first_name, last_name
        HAVING COUNT(*) > 1;

    c)  SELECT first_name, last_name, b.room_id AS "Room Number", COUNT(*) count
        FROM guest AS g 
        FULL JOIN booking AS b ON b.guest_id = g.id
        GROUP BY b.room_id, first_name, last_name
        HAVING COUNT(*) > 1;

    d)  SELECT DISTINCT b.room_id AS "Room Number", COUNT(*) AS "Number of Guests"
        FROM guest AS g 
        FULL JOIN booking AS b ON b.guest_id = g.id
        GROUP BY b.room_id
        HAVING COUNT(*) > 1;

3)
    a.  SELECT first_name AS "Student First Name", last_name AS "Student Last Name", c.subject AS "Class"
        FROM students AS s
        LEFT OUTER JOIN schedule AS sch on sch.student_id = s.id
        LEFT OUTER JOIN class AS c on c.id = sch.class_id
        WHERE c.subject = 'math';

    b.  SELECT grade, COUNT(*), c.subject
        FROM class AS c
        LEFT OUTER JOIN schedule AS s ON s.class_id = c.id
        GROUP BY grade, c.subject;

    c.  SELECT c.subject, COUNT(*) AS "Number of Students"
        FROM class AS c
        LEFT OUTER JOIN schedule AS s ON s.class_id = c.id
        LEFT OUTER JOIN students AS st ON st.id = s.student_id
        GROUP BY c.subject;

    d.  SELECT c.subject, COUNT(*) AS "Number of Students"
        FROM class AS c
        LEFT OUTER JOIN schedule AS s ON s.class_id = c.id
        LEFT OUTER JOIN students AS st ON st.id = s.student_id
        GROUP BY c.subject
        ORDER BY "Number of Students" DESC
        LIMIT 1;
